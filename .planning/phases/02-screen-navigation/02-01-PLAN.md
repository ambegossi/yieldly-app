    ```typescript
    export function formatAPY(apy: number): string {
      const abs = Math.abs(apy);
      const sign = apy < 0 ? "-" : "";

      if (abs >= 10000) {
        return `${sign}${(abs / 1000).toFixed(1)}K%`;
      }

      if (abs >= 1000) {
        return `${sign}${Math.round(abs).toLocaleString("en-US")}%`;
      }

      return `${apy.toFixed(2)}%`;
    }
    ```

    **2. Create `src/app/pool-details.tsx`** — route file (per D-04, D-06):

    ```typescript
    import { ScreenWrapper } from "@/components/screen-wrapper";
    import { type Pool } from "@/domain/pool/pool";
    import PoolDetailsScreen from "@/screens/pool-details";
    import { useLocalSearchParams, useRouter } from "expo-router";

    export default function PoolDetailsRoute() {
      const params = useLocalSearchParams<{
        id: string;
        chain: string;
        project: string;
        symbol: string;
        apy: string;
        url: string;
      }>();
      const router = useRouter();

      const pool: Pool = {
        id: params.id ?? "",
        chain: params.chain ?? "",
        project: params.project ?? "",
        symbol: params.symbol ?? "",
        apy: parseFloat(params.apy ?? "0"),
        url: params.url ?? "",
      };

      return (
        <ScreenWrapper>
          <PoolDetailsScreen pool={pool} onBack={() => router.back()} />
        </ScreenWrapper>
      );
    }
    ```

    Key details:
    - Flat route at `/pool-details` (not nested `pool/[id]`) — simpler since all data comes via params per D-04
    - `useLocalSearchParams` with typed generic for all 6 Pool fields
    - `apy` deserialized with `parseFloat()` (params are always strings — Pitfall 1)
    - Fallback `?? ""` / `?? "0"` for missing params
    - Wraps screen in `ScreenWrapper` (Suspense + ErrorBoundary) per existing pattern
    - Passes `pool` object and `onBack` callback (not the router object — per anti-pattern guidance)
    - DO NOT modify `_layout.tsx` — `headerShown: false` is already set per D-06
